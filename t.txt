 stage('List All Tools') {
            steps {
                 script {
                    def toolInstallationManager = Jenkins.instance.getExtensionList(hudson.model.JDK.DescriptorImpl.class)[0]
                    def allToolInstallations = toolInstallationManager.getInstallations()

                    echo "Available tools:"
                    allToolInstallations.each { toolInstallation ->
                        echo "  - ${toolInstallation.name} (${toolInstallation.home})"
                    }
                }
            }
        }
